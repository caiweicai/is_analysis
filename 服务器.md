    do { \  
    perror(m); \  
    exit(EXIT_FAILURE); \

void echo_ser(int sock)  
{  
    char recvbuf[1024] = {0};   //缓存数组，存放客户端接受的数据和信息
    struct sockaddr_in peeraddr;    //设置服务器地址信息
    socklen_t peerlen;  
    int n;  
      
    while (1)  
    {  
        /******************接收数据并返回数据**********************/
        peerlen = sizeof(peeraddr);  

        memset(recvbuf, 0, sizeof(recvbuf));  //刷新，

        n = recvfrom(sock, recvbuf, sizeof(recvbuf), 0,  
                     (struct sockaddr *)&peeraddr, &peerlen);    //从udp接收数据
        if (n <= 0)     //
        {  
              
            if (errno == EINTR)  
                continue;  
              
            ERR_EXIT("recvfrom error");  
        }  
        else if(n > 0)       //打印客户端的数据，并向客户端返回数据
        {  
            printf("接收到的数据：%s\n",recvbuf);  
            sendto(sock, recvbuf, n, 0,  
                   (struct sockaddr *)&peeraddr, peerlen);    //返回客户端数据
            printf("回送的数据：%s\n",recvbuf);  
        }  
    }  
    close(sock);  
}  
  
int main(void)  
{  
    int sock;  
//服务器建立socket，并描述
    if ((sock = socket(PF_INET, SOCK_DGRAM, 0)) < 0)  
        ERR_EXIT("socket error");  
      
//服务器地址信息
    struct sockaddr_in servaddr;  

//服务器填充sockaddr server结构
    memset(&servaddr, 0, sizeof(servaddr));  //刷新，将地址设置为0

    servaddr.sin_family = AF_INET;  //创建套接字时，用该字段指定地址家族，对于                                                       TCP/IP协议的，必须设置为AF_INET。

    servaddr.sin_port = htons(MYPORT);    //端口
    servaddr.sin_addr.s_addr = htonl(INADDR_ANY);  //地址
      
    printf("监听%d端口\n",MYPORT);  

//sock是socket的描述符，bind绑定sock
    if (bind(sock, (struct sockaddr *)&servaddr, sizeof(servaddr)) < 0)  
        ERR_EXIT("bind error");  
      
    echo_ser(sock);  
      
    return 0;  
}

/*
(1)建立套接字文件描述符，socket()；

(2)设置服务器地址和端口，struct sockaddr；

(3)向服务器发送数据，sendto()；

(4)接收服务器的数据，recvfrom()；

(5)关闭套接字，close()。

#include <unistd.h>  
#include <sys/types.h>  
#include <sys/socket.h>  
#include <netinet/in.h>  
#include <arpa/inet.h>  
#include <stdlib.h>  
#include <stdio.h>  
#include <errno.h>  
#include <string.h>  
  
#define MYPORT 8887    //发送端口
char* SERVERIP = "127.0.0.1";  //本地地址
  
#define ERR_EXIT(m) \  
    do \  
    { \  
    perror(m); \  
    exit(EXIT_FAILURE); \  
    } while(0)  
  
void echo_cli(int sock)  
{  
//设置发送服务器的地址
    struct sockaddr_in servaddr;  
    memset(&servaddr, 0, sizeof(servaddr));  
    servaddr.sin_family = AF_INET;  
    servaddr.sin_port = htons(MYPORT);  
    servaddr.sin_addr.s_addr = inet_addr(SERVERIP);  
      
    int ret;  
    char sendbuf[1024] = {0};  //发送缓存
    char recvbuf[1024] = {0};  //接收缓存
    while (fgets(sendbuf, sizeof(sendbuf), stdin) != NULL)  
    {  
          
        printf("向服务器发送：%s\n",sendbuf);  
        sendto(sock, sendbuf, strlen(sendbuf), 0, (struct sockaddr *)&servaddr, sizeof(servaddr));    //发送
          
        ret = recvfrom(sock, recvbuf, sizeof(recvbuf), 0, NULL, NULL);  
        if (ret == -1)  
        {  
            if (errno == EINTR)  
                continue;  
            ERR_EXIT("recvfrom");  
        }  
        printf("从服务器接收：%s\n",recvbuf);  
          
        memset(sendbuf, 0, sizeof(sendbuf));  
        memset(recvbuf, 0, sizeof(recvbuf));  
    }  
      
    close(sock);  
      
      
}  
  
int main(void)  
{  
    int sock;  //套接字描述
    if ((sock = socket(PF_INET, SOCK_DGRAM, 0)) < 0)  
        ERR_EXIT("socket");  
      
    echo_cli(sock);  
      
    return 0;  
}
